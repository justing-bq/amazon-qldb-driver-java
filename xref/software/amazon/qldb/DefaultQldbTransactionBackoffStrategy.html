14  package software.amazon.qldb;
15  
16  import java.time.Duration;
17  import java.util.Random;
18  import software.amazon.awssdk.utils.Validate;
19  
20  /**
21   * BackoffStrategy that calculates the time to delay the execution of the next transaction.
22   *
23   * <p>
24   * The backoff strategy uses a min delay of 10ms and max delay of 5000ms using an
25   * equal jitter strategy.
26   * </p>
27   *
28   * <pre>{@code
29   * exponential delay  = min( base delay * (2 ^ retry attempts), max delay)
30   * jitter = random()
31   * sleep time = exponential delay / 2 + jitter * exponential delay / 2
32   *
33   * }</pre>
34   */
35  public class DefaultQldbTransactionBackoffStrategy implements BackoffStrategy {
36      static final Duration BASE_DELAY_CEILING = Duration.ofMillis(10);
37      private static final Duration MAX_BACKOFF_CEILING = Duration.ofMillis(5000);
38      /**
39       * Max permitted retry times. To prevent exponentialDelay from overflow, there must be 2 ^ retriesAttempted
40       * <= 2 ^ 31 - 1, which means retriesAttempted <= 30, so that is the ceil for retriesAttempted.
41       */
42      private static final int RETRIES_ATTEMPTED_CEILING = (int) Math.floor(Math.log(Integer.MAX_VALUE) / Math.log(2));
43  
44      private final Duration baseDelay ;
45      private final Duration maxBackoffTime;
46      private final Random random;
47  
48  
49      DefaultQldbTransactionBackoffStrategy() {
50          this(BASE_DELAY_CEILING, MAX_BACKOFF_CEILING, new Random());
51      }
52  
53      DefaultQldbTransactionBackoffStrategy(final Duration baseDelay, final Duration maxBackoffTime, final Random random) {
54          this.baseDelay = Validate.isNotNegative(baseDelay, "baseDelay");
55          this.maxBackoffTime = Validate.isNotNegative(maxBackoffTime, "maxBackoffTime");
56  
57          if (baseDelay.compareTo(maxBackoffTime) > 0) {
58              throw new IllegalArgumentException(String.format("%s cannot be greater than %s", "baseDelay", "maxBackoffTime"));
59          }
60  
61          this.random = random;
62      }
63  
64      /**
65       * Constructor to create a Backoff Strategy but with custom baseDelay and cap backoff time.
66       * @param baseDelay The minimum amount of time to delay a retry.
67       * @param maxBackoffTime The maximum time to delay a retry.
68       */
69      public DefaultQldbTransactionBackoffStrategy(final Duration baseDelay, final Duration maxBackoffTime) {
70          this(baseDelay, maxBackoffTime, new Random());
71      }
72  
73      /**
74       * <p>Compute the delay before the next retry request.</p>
75       *
76       * <p>This strategy is only consulted when there will be a next retry.</p>
77       *
78       * @param retryPolicyContext Number of retries that will be attempted.
79       * @return Amount of time in milliseconds to wait before the next attempt. Must be non-negative (can be zero).
80       */
81      @Override
82      public Duration calculateDelay(RetryPolicyContext retryPolicyContext) {
83          int cappedRetries = Math.min(retryPolicyContext.retriesAttempted(), RETRIES_ATTEMPTED_CEILING);
84          int delay = (int) Math.min(baseDelay.multipliedBy(1L << cappedRetries).toMillis(), maxBackoffTime.toMillis());
85          return Duration.ofMillis((delay / 2) + random.nextInt((delay / 2) + 1));
86      }
87  
88  }
