31  /**
32   * Used to retrieve the results from QLDB, either asynchronously or synchronously.
33   *
34   * If configured to read asynchronously, a thread will be created which will read ahead of the current position and
35   * buffer chunks of data up to the configured readAhead setting. {@link #next()} will pause if the retrieval thread is
36   * still retrieving the next chunk of data when it is needed.
37   *
38   * If configured to read synchronously, then the next chunk of data will be retrieved from QLDB during the call to
39   * {@link #next()} when it has exhausted the current chunk of data.
40   */
41  class ResultRetriever {
42      private static final Logger logger = LoggerFactory.getLogger(ResultRetriever.class);
43  
44      private final Session session;
45      private Page currentPage;
46      private int currentResultValueIndex;
47  
48      private final Retriever retriever;
49      private final IonSystem ionSystem;
50      private final ExecutorService executorService;
51      private final AtomicBoolean isClosed;
52  
53      /**
54       * Constructor.
55       *
56       * @param session
57       *              The parent session that represents the communication channel to QLDB.
58       * @param firstPage
59       *              The first chunk of the result, returned by the initial execution.
60       * @param txnId
61       *              The unique ID of the transaction.
62       * @param readAhead
63       *              The number of buffers to asynchronously read ahead, 0 for synchronous retrieval.
64       * @param executorService
65       *              The executor service to use for asynchronous retrieval. Null if new threads should be created.
66       */
67      ResultRetriever(Session session, Page firstPage, String txnId, int readAhead, IonSystem ionSystem,
68                             ExecutorService executorService) {
69          Validate.isNotNegative(readAhead, "readAhead");
70  
71          this.session = session;
72          this.currentPage = firstPage;
73          this.currentResultValueIndex = 0;
74          this.ionSystem = ionSystem;
75          this.executorService = executorService;
76          this.isClosed = new AtomicBoolean(false);
77  
78          // Start the retriever thread if there are more chunks to retrieve.
79          if (currentPage.nextPageToken() == null) {
80              this.retriever = null;
81          } else if (0 == readAhead) {
82              this.retriever = new Retriever(session, txnId, currentPage.nextPageToken());
83          } else {
84              final ResultRetrieverRunnable runner = new ResultRetrieverRunnable(session, txnId,
85                      currentPage.nextPageToken(), readAhead, isClosed);
86              this.retriever = runner;
87  
88              if (null == executorService) {
89                  final Thread retrieverThread = new Thread(runner, "ResultRetriever");
90                  retrieverThread.setDaemon(true);
91                  retrieverThread.start();
92              } else {
93                  this.executorService.submit(runner);
94              }
95          }
96      }
97  
98      /**
99       * Indicate if there are any more data to be retrieved.
100      *
101      * @return {@code true} if there is more data and {@link #next()} will return an IonValue; {@code false} otherwise.
102      */
103     public synchronized boolean hasNext() {
104         if (isClosed.get()) {
105             throw QldbDriverException.create(Errors.RESULT_PARENT_INACTIVE.get(), retriever.txnId);
106         }
107         while (currentResultValueIndex >= currentPage.values().size()) {
108             if (null == currentPage.nextPageToken()) {
109                 return false;
110             }
111             currentPage = retriever.getNextPage();
112             currentResultValueIndex = 0;
113         }
114         return true;
115     }
116 
117     /**
118      * Retrieve the next IonValue in the result. Note that this should only be called if {@link #hasNext()} returns
119      * true.
120      *
121      * @return The next IonValue in the result.
122      * @throws NoSuchElementException if the iteration has no more elements.
123      */
124     public synchronized IonValue next() {
125         if (!hasNext()) {
126             throw new NoSuchElementException();
127         }
128         // TODO: Use an InputStream instead after the GitHub feature https://github.com/amzn/ion-java/issues/292 is resolved
129         final byte[] bytes = currentPage.values().get(currentResultValueIndex++).ionBinary().asByteArray();
130         return ionSystem.singleValue(bytes);
131     }
132 
133     /**
134      * Set this retriever to closed, indicating the parent transaction is closed.
135      */
136     void close() {
137         isClosed.set(true);
138     }

140     /**
141      * Basic object for retrieving the next chunk of data from QLDB for a result set.
142      */
143     private static class Retriever {
144         final Session session;
145         String nextPageToken;
146         private final String txnId;
147 
148         /**
149          * Constructor for creating the retriever for a specific session and result.
150          *
151          * @param session
152          *              The parent session to use in retrieving results.
153          * @param txnId
154          *              The unique ID of the transaction.
155          * @param nextPageToken
156          *              The unique token identifying the next chunk of data to fetch for this result set.
157          */
158         private Retriever(Session session, String txnId, String nextPageToken) {
159             this.session = session;
160             this.txnId = txnId;
161             this.nextPageToken = nextPageToken;
162         }
163 
164         /**
165          * Retrieve the next chunk of data from QLDB.
166          *
167          * @return The next chunk of data from QLDB.
168          * @throws QldbDriverException if an unexpected error occurs during result retrieval.
169          */
170         Page getNextPage() {
171             final Page result = session.sendFetchPage(txnId, nextPageToken).page();
172             nextPageToken = result.nextPageToken();
173             return result;
174         }
175     }
176 
177     private static class ResultRetrieverRunnable extends Retriever implements Runnable {
178         private final BlockingDeque<ResultHolder<Exception>> results;
179         private final int readAhead;
180         private final AtomicBoolean isClosed;
181 
182         /**
183          * Constructor for the Runnable responsible for asynchronous retrieval of results from QLDB.
184          *
185          * @param session
186          *              The parent session to use in retrieving results.
187          * @param txnId
188          *              The unique ID of the transaction.
189          * @param nextPageToken
190          *              The unique token identifying the next chunk of data to fetch for this result set.
191          * @param readAhead
192          *              The maximum number of chunks of data to buffer at any one time.
193          * @param isClosed
194          *              {@link AtomicBoolean} tracking the state of the parent {@link ResultRetriever}.
195          */
196         ResultRetrieverRunnable(Session session, String txnId, String nextPageToken, int readAhead,
197                                 AtomicBoolean isClosed) {
198             super(session, txnId, nextPageToken);
199             this.readAhead = Math.min(1, readAhead - 1);
200             this.results = new LinkedBlockingDeque<>(readAhead);
201             this.isClosed = isClosed;
202         }
203 
204         @Override
205         public void run() {
206             try {
207                 while (null != nextPageToken) {
208                     final Page page = super.getNextPage();
209                     try {
210                         while (!results.offer(new ResultHolder(page), 50, TimeUnit.MILLISECONDS)) {
211                             if (isClosed.get()) {
212                                 throw QldbDriverException.create(Errors.RESULT_PARENT_INACTIVE.get(),
213                                                                  super.txnId);
214                             }
215                             Thread.yield();
216                         }
217                     } catch (InterruptedException ie) {
218                         Thread.currentThread().interrupt();
219                         throw QldbDriverException.create(Errors.RETRIEVE_INTERRUPTED.get(), ie);
220                     }
221                 }
222             } catch (Exception e) {
223                 results.clear();
224                 if (!results.offerFirst(new ResultHolder<>(e))) {
225                     // We've failed to give back the exception; log it as a best case fallback.
226                     logger.error(String.format(Errors.QUEUE_CAPACITY.get(), readAhead), e);
227                 }
228             }
229         }
230 
231         @Override
232         Page getNextPage() {
233             try {
234                 final ResultHolder<Exception> result = results.take();
235                 if (null != result.getAssociatedValue()) {
236                     if (result.getAssociatedValue() instanceof RuntimeException) {
237                         throw (RuntimeException) result.getAssociatedValue();
238                     }
239                     throw new RuntimeException(result.getAssociatedValue());
240                 }
241 
242                 return result.getResult();
243             } catch (InterruptedException ie) {
244                 Thread.currentThread().interrupt();
245                 throw QldbDriverException.create(Errors.RETRIEVE_INTERRUPTED.get(), ie);
246             }
247         }
248     }
