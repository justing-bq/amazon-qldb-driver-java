14  package software.amazon.qldb;
15  
16  import java.time.Duration;
17  
18  /**
19   * <p>Define the Backoff strategy that is going to be used to retry the Transaction.</p>
20   *
21   * @see DefaultQldbTransactionBackoffStrategy to see the default implementation that is used if no Strategy
22   * is defined in the {@link QldbDriverBuilder}.
23   */
24  public interface BackoffStrategy {
25      /**
26       * Compute the delay before the next retry request. This strategy is only consulted when there will be a next retry.
27       *
28       * @param retryPolicyContext Context about the state of the last request and information about the number of requests made.
29       * @return Amount of time in milliseconds to wait before the next attempt. Must be non-negative (can be zero).
30       */
31      Duration calculateDelay(RetryPolicyContext retryPolicyContext);
32  }
