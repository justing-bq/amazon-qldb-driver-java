14  package software.amazon.qldb.exceptions;
15  
16  import java.util.ResourceBundle;
17  
18  /**
19   * Enum identifying the possible error types when executing a transaction.
20   *
21   * <p><b>Note</b>: this enum is for internal use only.</p>
22   */
23  public enum Errors {
24      ABORTED_EXCEPTION,         // txn aborted
25      GET_SESSION_INTERRUPTED, // illegal state
26      DRIVER_CLOSED,           // illegal state
27      QUEUE_CAPACITY,
28      RETRIEVE_INTERRUPTED,   // illegal state
29      RESULT_PARENT_INACTIVE, // illegal state
30      STREAM_RESULT_ITERATED, // illegal state
31  
32      NO_SESSION_AVAILABLE,   // driver problem
33      TXN_DIGEST_MISMATCH,    // driver problem
34      INCORRECT_TYPE,         // driver problem
35      SERIALIZING_PARAMS,     // driver problem
36  
37      GENERIC_EXCEPTION;
38  
39  
40      private static final ResourceBundle MESSAGES = ResourceBundle.getBundle("errors");
41  
42      /**
43       * Retrieve the localized error message associated with the enum value.
44       *
45       * @return The associated localized error message.
46       */
47      public String get() {
48          return MESSAGES.getString(this.name());
49      }
50  }
