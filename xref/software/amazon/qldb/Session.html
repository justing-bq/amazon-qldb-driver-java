52  /**
53   * Session object representing a communication channel with QLDB.
54   *
55   * This object is thread-safe.
56   */
57  @ThreadSafe
58  class Session implements AutoCloseable {
59      private static final Logger logger = LoggerFactory.getLogger(Session.class);
60  
61      private final String ledgerName;
62      private final String sessionToken;
63      private final String sessionId;
64      private final QldbSessionClient client;
65  
66      /**
67       * Constructor for a session to a specific ledger.
68       #
69       * @param ledgerName
70       *              The name of the ledger to create a session to.
71       * @param sessionToken
72       *              The unique identifying token for this session to the QLDB.
73       * @param sessionId
74       *              The initial request ID for this session to QLDB.
75       * @param client
76       *              The low-level session used for communication with QLDB.
77       */
78      private Session(String ledgerName, String sessionToken, String sessionId, QldbSessionClient client) {
79          this.ledgerName = ledgerName;
80          this.client = client;
81          this.sessionToken = sessionToken;
82          this.sessionId = sessionId;
83      }
84  
85      /**
86       * Factory method for constructing a new Session, creating a new session to the QLDB on construction.
87       *
88       * @param ledgerName
89       *              The name of the ledger to create a session to.
90       * @param client
91       *              The low-level session used for communication with QLDB.
92       *
93       * @return A newly created {@link Session}.
94       */
95      static Session startSession(String ledgerName, QldbSessionClient client) {
96          final StartSessionRequest request = StartSessionRequest.builder().ledgerName(ledgerName).build();
97          final SendCommandRequest command = SendCommandRequest.builder().startSession(request).build();
98  
99          logger.debug("Sending start session request: {}", command);
100         final SendCommandResponse result = client.sendCommand(command);
101         final String sessionToken = result.startSession().sessionToken();
102         final String sessionId = result.responseMetadata().requestId();
103 
104         return new Session(ledgerName, sessionToken, sessionId, client);
105     }
106 
107     @Override
108     public void close() {
109         try {
110             sendEndSession();
111         } catch (SdkServiceException e) {
112             // We will only log issues closing the session, as QLDB will clean them up after a timeout.
113             logger.warn("Errors closing session: {}", e.getMessage(), e);
114         }
115     }

117     /**
118      * Get the session ID.

120      * @return This session's ID.

122     String getId() {
123         return sessionId;
124     }

126     /**
127      * Get the session token.
128      *
129      * @return This session's token.
130      */
131     String getToken() {
132         return sessionToken;
133     }

135     /**
136      * Send an abort request to QLDB, rolling back any active changes and closing any open results.
137      *
138      * @return The result of the abort transaction request.
139      */
140     AbortTransactionResult sendAbort() {
141         final AbortTransactionRequest request = AbortTransactionRequest.builder().build();
142 
143         SendCommandResponse result = send(SendCommandRequest.builder().abortTransaction(request));
144         return result.abortTransaction();
145     }
146 
147     /**
148      * Send a commit request to QLDB, committing any active changes and closing any open results.
149      *
150      * @param txnId
151      *              The unique ID of the transaction to commit.
152      * @param transactionDigest
153      *              The digest hash of the transaction to commit.
154      *
155      * @return The result of the commit transaction request.
156      * @throws OccConflictException if an OCC conflict has been detected within the transaction.
157      */
158     CommitTransactionResult sendCommit(String txnId, ByteBuffer transactionDigest) {
159         final CommitTransactionRequest request = CommitTransactionRequest.builder()
160              .transactionId(txnId)
161              .commitDigest(SdkBytes.fromByteBuffer(transactionDigest))
162              .build();
163 
164         SendCommandResponse result = send(SendCommandRequest.builder().commitTransaction(request));
165         return result.commitTransaction();
166     }
167 
168     /**
169      * Send an end session request to QLDB, closing all open results and transactions.
170      *
171      * @return The result of the end session request.
172      */
173     EndSessionResult sendEndSession() {
174         final EndSessionRequest request = EndSessionRequest.builder().build();
175 
176         SendCommandResponse result = send(SendCommandRequest.builder().endSession(request));
177         return result.endSession();
178     }
179 
180     /**
181      * Send an execute request with parameters to QLDB.
182      *
183      * @param statement
184      *              The PartiQL statement to execute.
185      * @param parameters
186      *              The parameters to use with the PartiQL statement for execution.
187      * @param txnId
188      *              The unique ID of the transaction to execute.
189      *
190      * @return The result of the execution, which contains a {@link Page} representing the first data chunk.
191      */
192     ExecuteStatementResult sendExecute(String statement, List<IonValue> parameters, String txnId) {
193         final List<ValueHolder> byteParameters = new ArrayList<>(parameters.size());
194 
195         if (!parameters.isEmpty()) {
196             try {
197                 final IonBinaryWriterBuilder builder = IonBinaryWriterBuilder.standard();
198                 final ByteArrayOutputStream stream = new ByteArrayOutputStream();
199                 final IonWriter writer = builder.build(stream);
200                 for (IonValue parameter : parameters) {
201                     parameter.writeTo(writer);
202                     writer.finish();
203                     final SdkBytes sdkBytes = SdkBytes.fromByteArray(stream.toByteArray());
204                     final ValueHolder value = ValueHolder.builder().ionBinary(sdkBytes).build();
205                     byteParameters.add(value);
206 
207                     // Reset the stream so that it can be re-used.
208                     stream.reset();
209                 }
210             } catch (IOException e) {
211                 throw QldbDriverException.create(String.format(Errors.SERIALIZING_PARAMS.get(), e.getMessage()),
212                                                  e);
213             }
214         }
215 
216         final ExecuteStatementRequest request = ExecuteStatementRequest.builder()
217                                                                        .statement(statement)
218                                                                        .parameters(byteParameters)
219                                                                        .transactionId(txnId)
220                                                                        .build();
221         SendCommandResponse result = send(SendCommandRequest.builder().executeStatement(request));
222         return result.executeStatement();
223     }
224 
225     /**
226      * Send a fetch result request to QLDB, retrieving the next chunk of data for the result.
227 
228      * @param txnId
229      *              The unique ID of the transaction to execute.
230      * @param nextPageToken
231      *              The token that indicates what the next expected page is.
232      *
233      * @return The result of the fetch page request.
234      */
235     FetchPageResult sendFetchPage(String txnId, String nextPageToken) {
236         final FetchPageRequest request = FetchPageRequest.builder()
237                                                          .transactionId(txnId)
238                                                          .nextPageToken(nextPageToken)
239                                                          .build();
240 
241         SendCommandResponse result = send(SendCommandRequest.builder().fetchPage(request));
242         return result.fetchPage();
243     }
244 
245     /**
246      * Send a start transaction request to QLDB.
247      *
248      * @return The result of the start transaction request.
249      */
250     StartTransactionResult sendStartTransaction() {
251         final StartTransactionRequest request = StartTransactionRequest.builder().build();
252         final SendCommandRequest.Builder command = SendCommandRequest.builder().startTransaction(request);
253 
254         final SendCommandResponse result = send(command);
255         return result.startTransaction();
256     }
257 
258     /**
259      * Send a request to QLDB.
260      *
261      * @param request
262      *              The request to send.
263      *
264      * @return The result returned by QLDB for the request.
265      * @throws OccConflictException if an OCC conflict was detected when committing a transaction.
266      * @throws InvalidSessionException when this session is invalid.
267      */
268     private SendCommandResponse send(SendCommandRequest.Builder request) {
269         final SendCommandRequest command = request.sessionToken(sessionToken).build();
270         logger.debug("Sending request: {}", command);
271         return client.sendCommand(command);
272     }
