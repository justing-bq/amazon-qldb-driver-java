package software.amazon.qldb;

import software.amazon.awssdk.core.exception.SdkException;

/**
 * Contains useful information about a failed transaction that can be used to make retry and backoff decisions.
 * See {@link RetryPolicy}.
 */
public final class RetryPolicyContext {
    private final SdkException exception;
    private final int retriesAttempted;
    private final String transactionId;

    public RetryPolicyContext(SdkException exception, int retriesAttempted, String transactionId) {
        this.exception = exception;
        this.retriesAttempted = retriesAttempted;
        this.transactionId = transactionId;
    }

    public SdkException exception() {
        return exception;
    }

    public int retriesAttempted() {
        return retriesAttempted;
    }

    public String transactionId() {
        return transactionId;
    }

}
