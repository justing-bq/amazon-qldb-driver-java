213     @Test
214     public void testSendExecuteRaisesException() {
215         Mockito.when(mockSendCommandResult.executeStatement()).thenReturn(mockExecuteStatementResult);
216         Mockito.doAnswer(new Answer<String>() {
217             @Override
218             public String answer(InvocationOnMock invocation) throws Throwable {
219                 throw new IOException("msg");
220             }
221         }).when(mockIonValue).writeTo(ArgumentMatchers.any(IonWriter.class));
222 
223         final List<IonValue> MOCK_INVALID_PARAMETERS = Collections.singletonList(mockIonValue);
224 
225         final Session session = Session.startSession(MOCK_LEDGER_NAME, mockClient);
226 
227         assertThrows(QldbDriverException.class, () -> {
228             try {
229                 session.sendExecute(MOCK_STATEMENT, MOCK_INVALID_PARAMETERS, MOCK_TXN_ID);
230             } finally {
231                 Mockito.verify(mockClient, Mockito.times(1))
232                        .sendCommand(ArgumentMatchers.any(SendCommandRequest.class));
233             }
234         });
235     }
